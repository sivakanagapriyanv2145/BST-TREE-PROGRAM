
Create the Binary Search Tree for the following data and do apost order traversal ,

collect the input until ‘-1’ occurs.

Input Format

Input consists of data that is nodes of the tree.

Constraints

Fetch Input until -1 encounters.

Output Format

Display the constructed tree in post order traversal

Sample Input 0

10
4
12
1
6
17
11
-1
Sample Output 0

Post order : 1 6 4 11 17 12 10
#include<stdio.h>
#include<stdlib.h>
struct node{
    int data;
    struct node *left,*right;
};
struct node *root,*newnode,*head,*temp;
struct node *create(int x)
{
    newnode=(struct node *)malloc(sizeof(struct node));
    newnode->data=x;
    newnode->left=NULL;
    newnode->right=NULL;
    return newnode;
}
struct node *connect(struct node *root,int x)
{
    if(root==NULL)
    {
        root=create(x);
        return root;
    }
    if(x<root->data)
    {
        root->left=connect(root->left,x);
    }
    else if(x>root->data)
    {
        root->right=connect(root->right,x);
    }
    return root;
}
void display(struct node *root)
{
    
    if(root!=NULL)
    {
        display(root->left);
        display(root->right);
        printf("%d ",root->data);
    }
}
int main()
{
    int x;
    scanf("%d",&x);
    while(x>-1)
    {
        root=connect(root,x);
        scanf("%d",&x);
    }
    printf("Post order : ");
    display(root);
}
